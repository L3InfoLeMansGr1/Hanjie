def copyFrozen
        newG = Grid.new(@rows.size, @cols.size)
        @rows.zip(newG.rows).each {|oldRow, newRow|
                oldRow.zip(newRow).each { |oldCell, newCell|
                        newCell.frozenOf(oldCell)
                }
        }
        return newG
end
