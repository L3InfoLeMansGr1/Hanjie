def self.getAGrid(te,fif,tw,diff)

        tab = [te,fif,tw].reject{|i| i.grade == -1}

        # Same difficulty (not -1)
        if te == fif && fif == tw then
                case diff
                when :easy
                        return te
                when :intermediate
                        return fif
                when :hard
                        return tw
                end
        # Different difficulty
        elsif te != fif && fif != tw then
                case diff
                when :easy
                        return tab.min
                when :intermediate
                        if tab.size == 3 then return tab.sort[1]
                        else return tab.sort[0]
                        end
                when :hard
                        return tab.max
                end
        # 2 same difficulty
        elsif !tab.empty? then
                case diff
                when :easy
                        return tab.uniq.min
                when :intermediate
                        return tab.uniq.min
                when :hard
                        return tab.max
                end
        # Impossible way
        else
                get(diff)
        end
end
