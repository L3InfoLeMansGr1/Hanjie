def toBoolean
        # image --> [true,true,false,true]                                    true = blanc
        #                     [false,true,false,false]                          false = noir
        #                     [true,true,true,true]

        grid = []
        image = toGrey(@precision)
        image.each_pixel{|pixel| grid << !pixel.isBlack?}
        grid.each_slice(image.columns).to_a
end
