def donneNom(unEntier)
        tailleChemin = @chemin.bytesize
        tailleComplete = @listeFichiers[unEntier].bytesize
        @listeFichiers[unEntier][tailleChemin ... tailleComplete+1] #on ajoute 1 pour le "*"
end
