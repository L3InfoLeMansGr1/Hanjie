def setup_tree_view(treeview)
        renderer = Gtk::CellRendererText.new
        column   = Gtk::TreeViewColumn.new("Nom", renderer,  :text => 0)
        treeview.append_column(column)
        renderer = Gtk::CellRendererText.new
        column   = Gtk::TreeViewColumn.new("Score", renderer, :text => 1)
        treeview.append_column(column)
end
