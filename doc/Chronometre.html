    Compte les secondes et met le label référencé à jour

    # File Game/Core/Chronometre.rb, line 76
def run
        # garder en mémoire les dixièmes de seconde permet de diminuer la perte de temps induite par la pause.
                if @mode!=1
                        @sec += 1
                else
                        @sec -= 1
                        if(@sec <= 0)
                                stop
                                notifyNoMoreTime
                        end
                end
end
