def initialize(cells, fState, type)
        @cellsPos = []
        cells.each{ |cell|
                @cellsPos.push({"x"=>cell.row,"y"=>cell.col})
        }
        @firstState = fState
        @type = type
end

def replay(game)
        game.currentGuess.moves.add self
        @cellsPos.each{ |cell|
                if @type == :primary
                        game.cellAt(cell["x"],cell["y"]).primaryChange
                else
                        game.cellAt(cell["x"],cell["y"]).secondaryChange
                end
        }
end
