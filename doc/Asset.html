def delImg(widget)
        widget.each { |img|
                @images << img
                widget.remove(img)
        }
        self
end

def getImg
        if @images.size == 0
                # puts "création d'une nouvelle image #{@file}"
                return Gtk::Image.new(pixbuf: @buffer)
        end

        # puts "réutilisation d'une image #{@file}"
        return @images.shift
end
